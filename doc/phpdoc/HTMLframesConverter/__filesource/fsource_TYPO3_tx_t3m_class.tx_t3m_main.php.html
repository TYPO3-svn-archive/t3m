  228.     function handlePostVars()    {
  229.         // simply taken from tcdirectmail
  230.         if ($_REQUEST['send_now']) {
  231.             $GLOBALS['TYPO3_DB']->sql_query('UPDATE pages SET tx_tcdirectmail_senttime = '.time().' WHERE uid = '.$_REQUEST['id']);
  232.         }
  233.         if ($_REQUEST['send_test']) {
  234.             $rs = $GLOBALS['TYPO3_DB']->exec_SELECTquery('*', 'pages', 'uid = '.$_REQUEST['id']);
  235.             tx_tcdirectmail_tools::mailForTest($GLOBALS['TYPO3_DB']->sql_fetch_assoc($rs));
  236.         }
  237.         if ($_REQUEST['check_for_spam']) {
  238.             $content = tx_t3m_spam::checkForSpam($_REQUEST['id']);
  239.             $this->myFooter=$content; //$this->doc->section('',$content,0,1);
  240.         }
  241.         if ($_REQUEST['targetgroup']) {
  242.             $content = '<h3>'.$GLOBALS['LANG']->getLL('targetgroupusers').' \''.tx_t3m_addresses::getTargetGroupName($_REQUEST['targetgroup']).'\'</h3>';
  243.             $content .= tx_t3m_addresses::tableForTargetgroupUsers($_REQUEST['targetgroup']);
  244.             $this->myFooter=$content; //$this->doc->section('',$content,0,1);
  245.         }
  246.         if ($_REQUEST['receivergroup']) {
  247.             $content = '<h3>'.$GLOBALS['LANG']->getLL('receivergroupusers').'  \''.tx_t3m_addresses::getReceiverGroupName($_REQUEST['receivergroup']).'\'</h3>';
  248.             $content .= tx_t3m_addresses::tableForReceivers($_REQUEST['receivergroup']);
  249.             $this->myFooter=$content; //$this->doc->section('',$content,0,1);
  250.         }
  251.     }
