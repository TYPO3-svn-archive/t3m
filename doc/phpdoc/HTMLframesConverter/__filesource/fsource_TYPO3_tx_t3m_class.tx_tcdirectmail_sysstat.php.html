  33. class tx_tcdirectmail_sysstat {
  34.  
  35.    /**
  36.    * Show some system settings
  37.    *
  38.    * @return    string    some system settings
  39.    */
  40.    function viewSysStatus() {
  41.        global $LANG;
  42.        global $TYPO3_DB;
  43.        global $TYPO3_CONF_VARS;
  44. //        global $ICON_PATH;
  45.        $ICON_PATH = $GLOBALS['BACK_PATH'].t3lib_extMgm::extRelPath('t3m').'gfx/';
  46.  
  47.       /*********************** cron-scripts **********************/
  48.       /* Check if the cronScripts are setup correctly is executable */
  49.       $theConf = unserialize($TYPO3_CONF_VARS['EXT']['extConf']['tcdirectmail']);
  50.       $correct_php = $theConf['path_to_php'];
  51.  
  52.       $content .= '<h3>'.$LANG->getLL('cronjobs').'</h3>';
  53.       if ($_REQUEST['try_correct_path']) {
  54.          foreach ($TYPO3_CONF_VARS['EXTCONF']['tcdirectmail']['cliScripts'] as $script => $dummy) {
  55.             /* Correct the sh-bang */
  56.             $mailer_lines = file($script);
  57.             $mailer_lines[0] = "#!$correct_php\n";
  58.             $fd = fopen($script, 'w');
  59.             fwrite($fd, implode("", $mailer_lines));
  60.             fclose ($fd);
  61.  
  62.             /* Set exe-bit */
  63.             list(,,$mode) = stat($script);
  64.             $mode |= 0110;
  65.             chmod($script, $mode);
  66.          }
  67.  
  68.          clearstatcache();
  69.       }
  70.  
  71.       /* Display if it's ok */
  72.       if (file_exists($correct_php)) {
  73.          $content .= str_replace('###PHP_PATH###', $correct_php, $LANG->getLL('current_php_path'));
  74.  
  75.          foreach ($TYPO3_CONF_VARS['EXTCONF']['tcdirectmail']['cliScripts'] as $script => $frequency) {
  76.             $mailer_lines = file($script);
  77.             $current_php = trim(str_replace('#!', '', $mailer_lines[0]));
  78.  
  79.             if ($correct_php == $current_php) {
  80.                /* There is no crontab..  dont explain..  just say 'ok' */
  81.                if ($frequency == '') {
  82.                   $content .= '<p><img src="'.$ICON_PATH.'icon_ok.gif" />'.$script.'</p>';
  83.                } else {
  84.                   $content .= '<p><img src="'.$ICON_PATH.'icon_ok.gif" />'.
  85.                     str_replace(array('###SCRIPT###', '###FREQUENCY###'),
  86.                                 array($script, $frequency),
  87.                                 $LANG->getLL('crontab_explain')).'</p>';
  88.                }
  89.             } else {
  90.                $script_errors = 1;
  91.                $content .= '<p><img src="'.$ICON_PATH.'icon_fatalerror.gif" />'.
  92.                     str_replace('###SCRIPT###', $script, $LANG->getLL('crontab_bad_php_path')).'</p>';
  93.             }
  94.          }
  95.       } else {
  96.          $content .= '<p><img src="'.$ICON_PATH.'icon_fatalerror.gif" />'.
  97.                str_replace('###PHP_PATH###', $correct_php, $LANG->getLL('php_not_found'));
  98.       }
  99.  
  100.       /* If any errors, show the correct button */
  101.       if ($script_errors) {
  102.          $content .= '<form>';
  103.          $content .= '<input type="hidden" name="id" value="'.$_REQUEST['id'].'" />';
  104.          $content .= '<p><input type="submit" name="try_correct_path" value="'.$LANG->getLL('try_correct_path').'" /></p>';
  105.          $content .= '</form>';
  106.       }
  107.  
  108.       /********************** Helper programs *************************/
  109.       /* Check if we have a Lynx-executable */
  110.       $content .= '<h3>'.$LANG->getLL('lynx_browser').'</h3>';
  111.       if (file_exists($theConf['path_to_lynx'])) {
  112.          $content .= '<p><img src="'.$ICON_PATH.'icon_ok.gif" />'.
  113.                   str_replace('###LYNX_PATH###', $theConf['path_to_lynx'],
  114.                   $LANG->getLL('lynx_found')).'</p>';
  115.       } else {
  116.          $content .= '<p><img src="'.$ICON_PATH.'icon_warning.gif" />'.
  117.                   str_replace('###LYNX_PATH###', $theConf['path_to_lynx'],
  118.                   $LANG->getLL('lynx_not_found')).'</p>';
  119.       }
  120.  
  121.       /* Check if we have a Fetchmail-executable */
  122.       $content .= '<h3>'.$LANG->getLL('fetchmail_program').'</h3>';
  123.       if (file_exists($theConf['path_to_fetchmail'])) {
  124.          $content .= '<p><img src="'.$ICON_PATH.'icon_ok.gif" />'.
  125.                   str_replace('###FETCHMAIL_PATH###', $theConf['path_to_fetchmail'],
  126.                   $LANG->getLL('fetchmail_found')).'</p>';
  127.       } else {
  128.          $content .= '<p><img src="'.$ICON_PATH.'icon_warning.gif" />'.
  129.                   str_replace('###FETCHMAIL_PATH###', $theConf['path_to_fetchmail'],
  130.                   $LANG->getLL('fetchmail_not_found')).'</p>';
  131.       }
  132.  
  133.       /* Any other checks provided externally */
  134.       if (is_array($TYPO3_CONF_VARS['EXTCONF']['tcdirectmail']['viewSysStatusHook'])) {
  135.          foreach ($TYPO3_CONF_VARS['EXTCONF']['tcdirectmail']['viewSysStatusHook'] as $_classRef) {
  136.             $_procObj = & t3lib_div::getUserObj($_classRef);
  137.             $content = $_procObj->viewSysStatusHook($content, $LANG, $this);
  138.          }
  139.       }
  140.  
  141.       return $content;
  142.    }
  143.  
  144. }
