  71.     function handlePostVars() {
  72.         $out = $GLOBALS['LANG']->getLL('performingPostinstalls');
  73.  
  74.         //getting old config
  75.         $settings = $this->myConf;
  76.  
  77.         t3lib_div::debug($settings);
  78.         if (t3lib_div::_GP('createSysfolders')) {
  79.             $out .= tx_t3m_postinstall::createSysfolders();
  80.         }
  81.         if (t3lib_div::_GP('createFegroups')) {
  82.             $out .= tx_t3m_postinstall::createFegroups(); // check srfeuserregister for 'pending' 'subscriptions', others: 'tests','deregistrations', 'blocked, 'softbounces', 'hardbounces'

  84.             $out .= tx_t3m_postinstall::createTargetgroups();
  85.         }
  86.         if (t3lib_div::_GP('createBeUsers')) {
  87.             $out .= tx_t3m_postinstall::createBeGroups();
  88.             $out .= tx_t3m_postinstall::createBeUsers(); //"admin, controller, author"
  89.         }
  90. //         if (t3lib_div::_GP('createTargetgroups')) {
  91. //
  92. //         }
  93.         if (t3lib_div::_GP('createAll')) {
  94.             $out = tx_t3m_postinstall::createSysfolders();
  95.             $out .= tx_t3m_postinstall::createFegroups();
  96.             $out .= tx_t3m_postinstall::createBeGroups();
  97.             $out .= tx_t3m_postinstall::createTargetgroups();
  98.         }
  99.  
  100.         //now $settings has new values
  101.         //write settings
  102.         $out .= tx_t3m_settings::changeExtensionSettings('t3m', $settings);
  103.  
  104.         return $out;
  105.     }
