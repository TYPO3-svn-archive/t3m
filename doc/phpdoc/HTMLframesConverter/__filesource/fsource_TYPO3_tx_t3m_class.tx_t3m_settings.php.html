  260.     function changeExtensionSettings($extKey, $settings) {
  261.  
  262.         // read old values
  263.         if (!($extKey)) {
  264.             $extKey = $this->extKey; //t3m
  265.         }

  269.  
  270.         // read in new values from created stuff:
  271.         if (!($settings)) {
  272.             $out = 'No settings found to change..<br/>';
  273.             return $out;
  274.         }
  275.  
  276.         foreach ($settings as $key => $val) {
  277.             $this->myConf[$key] = $val;
  278.         }
  279.  
  280.         // get it as a serialized array:
  281.         //$GLOBALS['TYPO3_CONF_VARS']['EXT']['extConf']['t3m'] = serialize($this->myConf);
  282.         // put it back into the serialized array
  283.         //$newConfig = unserialize($GLOBALS['TYPO3_CONF_VARS']['EXT']['extConf']['t3m']);
  284.  
  285.         // create a string for the config file
  286.         $newConfigString = '$TYPO3_CONF_VARS[\'EXT\'][\'extConf\'][\''.$extKey.'\'] = \''.serialize($this->myConf).'\';';
  287.         $newConfigStringConfigOnly = serialize($this->myConf);
  288.  
  289.         // write the config into the conf file:
  290. //         $filename = t3lib_extMgm::extPath($this->extKey).'ext_localconf.php'; // not used anymore because it creates a conflict with EM, EM does not read the values!!
  291.         $filename = constant('PATH_typo3conf').'localconf.php'; // sensible stuff!
  292.         $handle = fopen($filename, "r");
  293.         $contents = fread($handle, filesize($filename));
  294.  
  295.         // first do a simple check if we have a config
  296.         $needle = '$TYPO3_CONF_VARS[\'EXT\'][\'extConf\'][\''.$extKey.'\']';

  299.              $newContents = str_replace('?>',$newConfigString."\n?>",$contents);

  301.         } else { // config was found and has to be altered

  303.             $pattern = '/(\$TYPO3_CONF_VARS\[\'EXT\'\]\[\'extConf\'\]\[\''.$extKey.'\'\] = \')(.*)(\';)(.*)/'; // $2 will be old config

  306.             preg_match($pattern,substr($subject,strlen($pattern)), $matches); // find our config

  308.  
  309.              $newContents = str_replace($matches[2],$newConfigStringConfigOnly,$contents); // and replace it with new

  311.  
  312. //             $newContents = str_replace($matches[0],'',$contents); // or delete it
  313. //             $out .= 'found! remove!';
  314.         }
  315.         fclose($handle);
  316.  
  317.     // write new config file:

  319.         fwrite($handle,$newContents);
  320.         fclose($handle);
  321.  
  322. /*        $out.=$filename.'<p>changed from: '.htmlspecialchars($contents).'</p>';
  323.         $out.='<p>to:<br/>'.htmlspecialchars($newContents);
  324. */        $out.='<br/>Please clear typo3conf cache!!!</p>';
  325.  
  326.         return $out;
  327.     }
