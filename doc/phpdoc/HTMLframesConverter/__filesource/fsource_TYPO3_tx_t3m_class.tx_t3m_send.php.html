  91.                 $res2 = $GLOBALS['TYPO3_DB']->exec_SELECTquery(
  92.                     'pid,sendtime',
  93.                     'tx_tcdirectmail_sentlog',

  95.                 );

  97.                 if ($timessent > 0) {
